	desc = sweden_event.2.d
	flavor = sweden_event.2.f

	event_image = {
		video = "europenorthamerica_capitalists_meeting"
	}

	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	duration = 3


	trigger = {
		country_has_primary_culture = cu:swedish
		has_variable = mortala_verkstad
		has_technology_researched = labor_movement
		NOT = {
			has_variable = var_event_2
		}
	}

	immediate = {
        S:STATE_SVEALAND = {
			random_scope_state = {
				limit = {
					owner = {
						c:SWE = THIS
					}
				}
				save_scope_as = svea_state
			}
		}
		set_variable = var_event_2
	}

	placement = scope:svea_state

	option = {
		name = sweden_event.2.a
		default_option = yes
		scope:svea_state = {
		    add_modifier = {
				name = motala_strike_modi
				months = 6
			}
			add_loyalists = {
				value = 0.1
				pop_type = laborers
			}
		}
	}

	option = {
		name = sweden_event.2.b
		scope:svea_state = {
		    add_modifier = {
				name = motala_strike_modi
				months = 3
			}
			add_radicals = {
				value = 0.1
				pop_type = laborers
			}
		}
	}
}
